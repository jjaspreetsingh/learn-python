Basic Data Analysis

data_analysis.py

import pandas as pd
import numpy as np

# Load data
df = pd.read_csv('grades.csv')

# Display basic information
print("Dataset Info:")
print(df.info())
print("\nFirst 5 rows:")
print(df.head())

# Calculate averages
print("\nAverage scores:")
print(df[['Math', 'Science', 'English']].mean())

# Find highest scorer in each subject
print("\nHighest scores:")
for subject in ['Math', 'Science', 'English']:
    max_score = df[subject].max()
    student = df[df[subject] == max_score]['Name'].values[0]
    print(f"{subject}: {student} ({max_score})")

# Calculate overall average for each student
df['Overall'] = df[['Math', 'Science', 'English']].mean(axis=1)
print("\nOverall averages:")
print(df[['Name', 'Overall']])

Data Visualization

visualization.py

import matplotlib.pyplot as plt

# Bar chart of averages
subjects = ['Math', 'Science', 'English']
averages = df[subjects].mean()

plt.figure(figsize=(10, 6))
plt.bar(subjects, averages, color=['blue', 'green', 'red'])
plt.title('Average Scores by Subject')
plt.ylabel('Average Score')
plt.ylim(0, 100)

for i, v in enumerate(averages):
    plt.text(i, v + 1, f'{v:.1f}', ha='center')

plt.show()

# Scatter plot
plt.figure(figsize=(8, 6))
plt.scatter(df['Math'], df['Science'])
plt.title('Math vs Science Scores')
plt.xlabel('Math Score')
plt.ylabel('Science Score')

# Add student names as labels
for i, name in enumerate(df['Name']):
    plt.annotate(name, (df['Math'][i], df['Science'][i]))

plt.show()
