Python Cheat Sheets

Quick reference guides for Python syntax and common operations.

Basic Syntax

Variables and Data Types

# Variables
name = "Alice"
age = 25
height = 5.7
is_student = True

# Data Types
str_var = "Hello"      # String
int_var = 42          # Integer
float_var = 3.14      # Float
bool_var = True       # Boolean

# Type conversion
int("42")     # String to int
str(42)       # Int to string
float("3.14") # String to float

Control Flow

Conditionals

if condition:
    # code
elif another_condition:
    # code
else:
    # code

# Comparison operators: == != < > <= >=
# Logical operators: and or not

Loops

# For loop
for item in iterable:
    # code

# Range
for i in range(5):     # 0,1,2,3,4
for i in range(1,6):   # 1,2,3,4,5

# While loop
while condition:
    # code

# Loop control
break    # Exit loop
continue # Skip iteration

Functions

Function Definition

def function_name(param1, param2=default):
    # code
    return value

# Calling functions
result = function_name(arg1, arg2)

Data Structures

Lists

# Create
my_list = [1, 2, 3, "hello"]

# Access
my_list[0]      # First element
my_list[-1]     # Last element

# Modify
my_list.append(4)     # Add to end
my_list.insert(0, 0)  # Insert at index
my_list.remove(2)     # Remove value
del my_list[1]        # Remove by index

# List comprehension
squares = [x**2 for x in range(10)]

Dictionaries

# Create
my_dict = {"key": "value", "name": "Alice"}

# Access
my_dict["key"]
my_dict.get("key", "default")

# Modify
my_dict["new_key"] = "new_value"
my_dict.update({"key": "updated"})

# Methods
my_dict.keys()    # Get keys
my_dict.values()  # Get values
my_dict.items()   # Get key-value pairs

File Handling

Reading/Writing Files

# Read file
with open("file.txt", "r") as f:
    content = f.read()
    lines = f.readlines()

# Write file
with open("file.txt", "w") as f:
    f.write("Hello, World!")

# Append
with open("file.txt", "a") as f:
    f.write("More text")

Common String Methods

String Operations

text = "Hello, World!"

# Case conversion
text.upper()     # "HELLO, WORLD!"
text.lower()     # "hello, world!"
text.title()     # "Hello, World!"

# Searching
text.find("World")    # 7
"World" in text       # True

# Splitting/Joining
words = text.split()  # ["Hello,", "World!"]
" ".join(words)       # "Hello, World!"

# Formatting
f"My name is {name}"  # f-string
"My age is {}".format(age)
